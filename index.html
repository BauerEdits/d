from flask import Flask, render_template_string, request, redirect, url_for, session, flash from flask_sqlalchemy import SQLAlchemy from datetime import datetime import os app = Flask(__name__) app.secret_key = 'super-secret-key' # Change this in production! # Configure SQLite database basedir = os.path.abspath(os.path.dirname(__file__)) app.config['SQLALCHEMY_DATABASE_URI'] = 'sqlite:///' + os.path.join(basedir, 'app.db') app.config['SQLALCHEMY_TRACK_MODIFICATIONS'] = False db = SQLAlchemy(app) # Payout multipliers based on number of legs in a bet: MULTIPLIERS = { 2: 3, 3: 7, 4: 10, 5: 12, 6: 25 } # ------------------------------ # Database Models # ------------------------------ class User(db.Model): id = db.Column(db.Integer, primary_key=True) username = db.Column(db.String(80), unique=True, nullable=False) password = db.Column(db.String(80), nullable=False) # Plain text for demo purposes! role = db.Column(db.String(20), nullable=False) # "admin" or "member" class Line(db.Model): id = db.Column(db.Integer, primary_key=True) description = db.Column(db.String(200), nullable=False) # outcome: None = not yet settled; True = correct; False = incorrect outcome = db.Column(db.Boolean, nullable=True) class Bet(db.Model): id = db.Column(db.Integer, primary_key=True) user_id = db.Column(db.Integer, db.ForeignKey('user.id')) bet_amount = db.Column(db.Float, nullable=False) # Store the IDs of the selected lines as a commaâ€‘separated string (e.g. "1,3,5") selected_lines = db.Column(db.String(100), nullable=False) num_legs = db.Column(db.Integer, nullable=False) is_settled = db.Column(db.Boolean, default=False) is_won = db.Column(db.Boolean, nullable=True) payout = db.Column(db.Float, nullable=True) created_at = db.Column(db.DateTime, default=datetime.utcnow) resolved_at = db.Column(db.DateTime, nullable=True) user = db.relationship('User', backref='bets') # ------------------------------ # Initial Setup # ------------------------------ def init_db(): db.create_all() # Create default admin and member accounts if they do not exist. if not User.query.filter_by(username='admin').first(): admin = User(username='admin', password='admin', role='admin') db.session.add(admin) if not User.query.filter_by(username='member').first(): member = User(username='member', password='member', role='member') db.session.add(member) db.session.commit() init_db() # ------------------------------ # Helper Functions # ------------------------------ def login_required(role=None): """Decorator to require login and optionally a specific role.""" from functools import wraps def decorator(f): @wraps(f) def decorated_function(*args, **kwargs): if 'user_id' not in session: flash("You need to log in first.") return redirect(url_for('login')) if role and session.get('role') != role: flash("Unauthorized access.") return redirect(url_for('login')) return f(*args, **kwargs) return decorated_function return decorator def evaluate_bets(): """Check unsettled bets. If all outcomes for the lines in the bet are set, mark the bet as won (if every selected line is correct) or lost.""" unsettled_bets = Bet.query.filter_by(is_settled=False).all() for bet in unsettled_bets: # Convert the comma-separated string into a list of line IDs. line_ids = [int(x) for x in bet.selected_lines.split(',') if x] lines = Line.query.filter(Line.id.in_(line_ids)).all() # Only evaluate the bet if all selected lines have their outcomes set. if all(line.outcome is not None for line in lines) and len(lines) == bet.num_legs: if all(line.outcome for line in lines): bet.is_won = True multiplier = MULTIPLIERS.get(bet.num_legs, 1) bet.payout = bet.bet_amount * multiplier else: bet.is_won = False bet.payout = 0 bet.is_settled = True bet.resolved_at = datetime.utcnow() db.session.commit() # ------------------------------ # Routes # ------------------------------ @app.route('/') def home(): return redirect(url_for('login')) # ----- Login/Logout ----- @app.route('/login', methods=['GET', 'POST']) def login(): login_form = '''

Login

Username:
Password:
''' if request.method == 'POST': username = request.form.get('username') password = request.form.get('password') user = User.query.filter_by(username=username, password=password).first() if user: session['user_id'] = user.id session['username'] = user.username session['role'] = user.role flash("Logged in successfully as " + user.role) if user.role == 'admin': return redirect(url_for('admin_dashboard')) else: return redirect(url_for('member_dashboard')) else: flash("Invalid credentials. Try again.") return render_template_string(login_form) return render_template_string(login_form) @app.route('/logout') def logout(): session.clear() flash("Logged out successfully.") return redirect(url_for('login')) # ----- Admin Routes ----- @app.route('/admin/dashboard') @login_required(role='admin') def admin_dashboard(): lines = Line.query.all() bets = Bet.query.all() dashboard_html = '''

    • {% endfor %}

All Bets

    {% for bet in bets %}
  • Bet ID: {{ bet.id }} by User ID: {{ bet.user_id }}, Amount: {{ bet.bet_amount }}, Lines: {{ bet.selected_lines }}, Settled: {{ bet.is_settled }}, {% if bet.is_settled %} {% if bet.is_won %}Won! Payout: {{ bet.payout }}{% else %}Lost{% endif %} {% else %} Pending {% endif %}
    • {% endfor %}
''' return render_template_string(dashboard_html, lines=lines, bets=bets) @app.route('/admin/create_line', methods=['POST']) @login_required(role='admin') def create_line(): description = request.form.get('description') if description: new_line = Line(description=description) db.session.add(new_line) db.session.commit() flash("Line created successfully.") return redirect(url_for('admin_dashboard')) @app.route('/admin/delete_line/') @login_required(role='admin') def delete_line(line_id): line = Line.query.get_or_404(line_id) db.session.delete(line) db.session.commit() flash("Line deleted successfully.") return redirect(url_for('admin_dashboard')) @app.route('/admin/update_line/', methods=['GET', 'POST']) @login_required(role='admin') def update_line(line_id): line = Line.query.get_or_404(line_id) if request.method == 'POST': outcome = request.form.get('outcome') if outcome == 'correct': line.outcome = True elif outcome == 'incorrect': line.outcome = False db.session.commit() flash("Line outcome updated.") return redirect(url_for('admin_dashboard')) update_form = '''

Set Outcome for Line ID: {{ line.id }}

{{ line.description }}



Back to Dashboard

''' return render_template_string(update_form, line=line) @app.route('/admin/resolve_bets', methods=['POST']) @login_required(role='admin') def resolve_bets(): evaluate_bets() flash("Bets have been evaluated.") return redirect(url_for('admin_dashboard')) # ----- Member Routes ----- @app.route('/member/dashboard') @login_required(role='member') def member_dashboard(): # Only show lines that have not been settled so members can bet on them. available_lines = Line.query.filter_by(outcome=None).all() user_id = session['user_id'] live_bets = Bet.query.filter_by(user_id=user_id, is_settled=False).all() past_bets = Bet.query.filter_by(user_id=user_id, is_settled=True).all() member_html = '''

    • {% endfor %}

Past Bets

    {% for bet in past_bets %}
  • Bet ID: {{ bet.id }}, Amount: {{ bet.bet_amount }}, Lines: {{ bet.selected_lines }}, {% if bet.is_won %}Won! Payout: {{ bet.payout }}{% else %}Lost{% endif %}, Resolved on: {{ bet.resolved_at }}
    • {% endfor %}
''' return render_template_string(member_html, available_lines=available_lines, live_bets=live_bets, past_bets=past_bets, session=session) @app.route('/member/place_bet', methods=['POST']) @login_required(role='member') def place_bet(): line_ids = request.form.getlist('line_ids') bet_amount = request.form.get('bet_amount') if not line_ids or len(line_ids) < 2: flash("You must select at least 2 lines.") return redirect(url_for('member_dashboard')) if len(line_ids) > 6: flash("You cannot select more than 6 lines.") return redirect(url_for('member_dashboard')) try: bet_amount = float(bet_amount) except ValueError: flash("Invalid bet amount.") return redirect(url_for('member_dashboard')) new_bet = Bet( user_id=session['user_id'], bet_amount=bet_amount, selected_lines=",".join(line_ids), num_legs=len(line_ids) ) db.session.add(new_bet) db.session.commit() flash("Bet placed successfully.") return redirect(url_for('member_dashboard')) # ------------------------------ # Run the Application # ------------------------------ if __name__ == '__main__': app.run(debug=True)
